    48    /*! 
    49     *  ======== Agent ========
    50     *
    51     *  This module is deprecated and will not be supported in a future release.
    52     *  Please use the UIA product for the equivalent functionality.
    53     *
    54     *  The Agent module provides target support for the CCS RTA tools.
    55     *
    56     *  The Agent module includes configuration parameters to help set up the
    57     *  SYS/BIOS logging on the target.
    58     *  
    59     *  Earlier versions of the Agent provided support for transfer of data
    60     *  during runtime.   Runtime support has been removed and
    61     *  the Agent now only supports "stop mode" updates.   Log data is stored
    62     *  in buffers in RAM and only updated on the host with the target is halted.
    63     *
    64     *  Runtime support is now handled via the System Analyzer and UIA products.
    65     */
    66    metaonly module Agent
    67    {
    68        /*!
    69         *  ======== loadLogging ========
    70         *  Enable the Load module event logging.
    71         *
    72         *  If this is false, the events will be disabled and cannot be enabled
    73         *  at runtime (ALWAYS_OFF).
    74         *  If this is true, the events will be enabled and can also be modified
    75         *  at runtime (RUNTIME_ON).
    76         */
    77        metaonly config Bool loadLogging = true;
    78        
    79        /*!
    80         *  ======== loadLoggingRuntimeControl ========
    81         *  Specify whether load logging can be enabled / disabled at runtime.
    82         *
    83         *  This determines what diags settings are applied to the module's diags
    84         *  mask. If 'false', the diags bits will be configured as ALWAYS_OFF or 
    85         *  ALWAYS_ON, meaning they can't be changed at runtime. If 'true', the 
    86         *  bits will be configured as 'RUNTIME_OFF' or 'RUNTIME_ON'.
    87         */
    88        metaonly config Bool loadLoggingRuntimeControl = true;
    89        
    90        /*!
    91         *  ======== loadLoggerSize ========
    92         *  Size (in MAUs) of logger used for the Load module Log events.
    93         *
    94         *  To specify the size in number of records, instead of MAUs, use the
    95         *  following code:
    96         *  
    97         *  @p(code)
    98         *  var Agent = xdc.useModule('ti.sysbios.rta.Agent');
    99         *  var LoggerBuf = xdc.useModule('xdc.runtime.LoggerBuf');
   100         *  
   101         *  // Set the load logger size to 50 records.
   102         *  Agent.loadLoggerSize = 50 * LoggerBuf.Entry.$sizeof();
   103         *
   104         *  // Print the final size to see how much memory is used.
   105         *  print("Load Logger Size (MAUs): " + Agent.loadLoggerSize);
   106         *  @p
   107         */
   108        metaonly config SizeT loadLoggerSize = 512;
   109    
   110        /*!
   111         *  ======== loadLoggerBufferSection ========
   112         *  Section to place the Load LoggerBuf's buffer.
   113         */
   114        metaonly config String loadLoggerBufferSection = "[Default]";
   115            
   116        /*!
   117         *  ======== mainLogging ========
