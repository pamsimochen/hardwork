    39    import xdc.runtime.Error;

    41    /*!
    42     *  ======== DEV ========
    43     *  Device Manager.
    44     *
    45     *  The DEV Device Manager manages the table of IOM drivers. Drivers can
    46     *  be added or removed from the system using DEV_create() and DEV_delete().
    47     *
    48     *  DEV_create() calls mdBindDev() for the underlying IOM driver.  If 
    49     *  successful, the driver gets added to the driver table.  GIO_create()
    50     *  looks in this table for matching device name when opening an I/O
    51     *  channel.
    52     *
    53     *  DEV_delete() calls mdUnBindDev() for the underlying IOM driver and 
    54     *  removes the entry from the driver table.
    55     */
    56    
    57    @InstanceFinalize
    58    @InstanceInitError
    59    @ModuleStartup
    60    
    61    module DEV 
    62    {
    63        /*! Init function type definition. */
    64        typedef Void (*InitFxn)();
    65        
    66        /*!
    67         *  ======== Fxns ========
    68         *  @_nodoc
    69         *  This function table mirrors the IOM_Fxns table.  Fxns * is used in the
    70         *  instance object instead of Ptr to support static configuration.
    71         */
    72        struct Fxns {
    73            Fxn mdBindDev;
    74            Fxn mdUnBindDev;
    75            Fxn mdControlChan;
    76            Fxn mdCreateChan;
    77            Fxn mdDeleteChan;
    78            Fxn mdSubmitChan;
    79        };
    80        
    81        /*!
    82         *  ======== DeviceParams ========
    83         *  @_nodoc
    84         *  Dummy DeviceParams structure.  This is used in the instance object
    85         *  instead of Ptr to support static configuration.
    86         */
    87        struct DeviceParams {
    88            Int dummy;
    89        };
    90       
    91        /*! 
    92         *  Size of the device table.
    93         *
    94         *  This size needs to be large enough to hold the sum of the drivers
    95         *  created at configuration time and runtime.
    96         */
    97        config UInt tableSize = 8;
    98    
    99        metaonly struct BasicView {
   100            String          name;
   101            Ptr             fxns;
   102            String          initFxn[];
   103            Int             devid;
   104            Ptr             deviceParams;
   105            Ptr             devp; 
   106        };
   107    
   108        @Facet
   109        metaonly config ViewInfo.Instance rovViewInfo = 
   110            ViewInfo.create({
   111                viewMap: [
   112                    ['Basic', {type: ViewInfo.INSTANCE, viewInitFxn: 'viewInitBasic', structName: 'BasicView'}],
   113                ]
   114            });
   115    
   116        /*!
   117         *  ======== match ========
   118         *  DEV_match searches the device table for the first device name that
   119         *  matches a prefix of 'name'.  The output parameter, device, points to
   120         *  the appropriate entry in the device table if successful and is set to
   121         *  NULL on error.
   122         *
   123         *  The suffix string return value contains a pointer to the characters
   124         *  remaining after the match.  This string can be used to specify device
   125         *  parameters or the name of another device driver.
   126         *
   127         *  For example, if you have "/uart" in the device table, then the
   128         *  suffix return value for the following would return a pointer to "0".
   129         *
   130         *  @p(code)
   131         *  suffix = DEV_match("/uart0", &device);
   132         *  @p
   133         *
   134         *  @param(name)    name of device
   135         *  @param(device)  pointer to DEV Handle (output param)
   136         *  @b(returns)     pointer to remaining characters after match
   137         */
   138        @DirectCall
   139        String match(String name, Handle *device);
   140    
   141    instance:
   142    
   143        /*!
   144         *  ======== create ========
   145         *  Add an IOM driver to the device table at runtime.
   146         * 
   147         *  The 'name' parameter must be a static string since this name
   148         *  is referenced via a pointer and is not copied.  For example:
   149         *
   150         *  @p(code)
   151         *  Error_init(&eb);
   152         *  DEV_create("/a2d", &A2DFXNS, &myparams, &eb);
   153         *  @p
   154         *
   155         *  The following code will not work since 'localstring' is a local
   156         *  variable:
   157         *
   158         *  @p(code)
   159         *  { 
   160         *      Char localstring[10];
   161         *      strcpy(localstring, "/a2d");
   162         *      Error_init(&eb);
   163         *      DEV_create(localstring, &A2DFXNS, &myparams, &eb);
   164         *  }
   165         *  @p
   166         *
   167         *  @param(name)    name of device (must be a static string)
   168         *  @param(fxns)    pointer to IOM_Fxns table
   169         */
   170        @DirectCall
   171        create(String name, Ptr fxns);
   172    
   173        /*!
   174         *  ======== getName ========
   175         *  Get name of device
   176         *
   177         *  DEV_getName returns the name of the referenced device.
   178         *
   179         *  @b(returns)     device name
   180         */
   181        @DirectCall
   182        String getName();
   183    
   184        /*!
   185         *  ======== getFxns ========
   186         *  Get IOM function table of device
   187         *
   188         *  DEV_getFxns returns the IOM function table of the referenced device.
   189         *
   190         *  @b(returns)     pointer to device function table
   191         */
   192        @DirectCall
   193        Ptr getFxns();
   194    
   195        /*!
   196         *  ======== getInitFxn ========
   197         *  Get initialized function of device
   198         *
   199         *  DEV_getInitFxn returns the init function of the referenced device.
   200         *
   201         *  @b(returns)     init function pointer 
   202         */
   203        @DirectCall
   204        InitFxn getInitFxn();
   205    
   206        /*!
   207         *  ======== getDevid ========
   208         *  Get devid of device
   209         *
   210         *  DEV_getDevid returns the devid of the referenced device.
   211         *
   212         *  @b(returns)     devid
   213         */
   214        @DirectCall
   215        Int getDevid();
   216    
   217        /*!
   218         *  ======== getDeviceParams ========
   219         *  Get deviceParams of device
   220         *
   221         *  DEV_getDeviceParams returns the deviceParams of the referenced device.
   222         *
   223         *  @b(returns)     deviceParams
   224         */
   225        @DirectCall
   226        Ptr getDeviceParams();
   227    
   228        /*!
   229         *  ======== getDevp ========
   230         *  Get devp of device
   231         *
   232         *  DEV_getDevp returns the devp of the referenced device.
   233         *
   234         *  @b(returns)     devp
   235         */
   236        @DirectCall
   237        Ptr getDevp();
   238    
   239        /*!
   240         *  ======== deviceParams ========
   241         *  Device-specific parameters
   242         * 
   243         *  This parameter is passed as the last parameter to the underlying
   244         *  IOM driver's mdBindDevice function.
   245         */
   246        config Ptr deviceParams = null;
   247    
   248        /*!
   249         *  ======== initFxn ========
   250         *  Driver Initialization function
   251         *     
   252         *  This function is called once for every entry in the device table.
   253         *  If this parameter is set to NULL (the default), then no function
   254         *  will be called.
   255         */
   256        config InitFxn initFxn = null;
   257    
   258        /*!
   259         *  ======== devid ========
   260         *  Device id
   261         */
   262        config Int devid = 0;
   263    
   264    internal:
   265    
   266        Int postInit(Object *obj, Error.Block *eb);
   267    
   268        struct Instance_State {
   269            String          name;           /* device name */
   270            Fxns            *fxns;          /* pointer to IOM function table */ 
   271            InitFxn         initFxn;        /* driver init function */
   272            Int             devid;          /* device id */
   273            DeviceParams    *deviceParams;  /* device parameters */
   274            Ptr             devp;           /* pointer to device global data */
   275        };
   276    
   277        struct Module_State {
   278            Handle          table[];        /* device table */
   279        };
