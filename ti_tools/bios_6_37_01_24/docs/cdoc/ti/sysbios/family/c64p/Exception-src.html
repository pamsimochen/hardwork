    44    /*!
    45     *  ======== Exception ========
    46     *  Exception Module
    47     *
    48     *  The Exception module is a basic C64+ exception handler.  It is generally
    49     *  considered to be a program endpoint, since an exception usually
    50     *  indicates something fatal to the system.
    51     *
    52     *  During initialization, the Exception module sets TSR.GEE and TSR.XEN to
    53     *  enable the CPU to recognize the EXECP input.
    54     *
    55     *  Function hooks are provided to the user for hooking in their own functions
    56     *  at different points of an exception. The hook functions are called in the
    57     *  following order:
    58     *  (1) exceptionHook - called whenever an exception occurs.
    59     *  (2) internalHook - called only when an internal exception occurs.
    60     *  (3) externalHook - called only when an external exception occurs.
    61     *  (4) nmiHook - called only when a legacy NMI occurs.
    62     *  (5) returnHook - called whenever an exception occurs.
    63     *
    64     *  @p(html)
    65     *  <h3> Calling Context </h3>
    66     *  <table border="1" cellpadding="3">
    67     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center"></colgroup>
    68     *
    69     *    <tr><th> Function                 </th><th>  Hwi   </th><th>  Swi   </th><th>  Task  </th><th>  Main  </th><th>  Startup  </th></tr>
    70     *    <!--                                                                                                                 -->
    71     *    <tr><td> {@link #clearLastStatus}         </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    72     *    <tr><td> {@link #evtEvtClear}             </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    73     *    <tr><td> {@link #evtExpMaskEnable}        </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    74     *    <tr><td> {@link #getLastStatus}           </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    75     *    <tr><td> {@link #setReturnPtr}            </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
    76     *    <tr><td colspan="6"> Definitions: <br />
    77     *       <ul>
    78     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
    79     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
    80     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
    81     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
    82     *           <ul>
    83     *             <li> In your module startup after this module is started (e.g. Mod_Module_startupDone() returns TRUE). </li>
    84     *             <li> During xdc.runtime.Startup.lastFxns. </li>
    85     *             <li> During main().</li>
    86     *             <li> During BIOS.startupFxns.</li>
    87     *           </ul>
    88     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
    89     *           <ul>
    90     *             <li> During xdc.runtime.Startup.firstFxns.</li>
    91     *             <li> In your module startup before this module is started (e.g. Mod_Module_startupDone() returns FALSE).</li>
    92     *           </ul>
    93     *       </ul>
    94     *    </td></tr>
    95     *
    96     *  </table>
    97     *  @p
    98     */

   210        /*! Status - structure filled by getLastStatus(). */
   211        struct Status {
   212            Bits32  efr;            //! Exception flag register
   213            Bits32  nrp;            //! NMI return pointer register
   214            Bits32  ntsr;           //! NMI/Exception task state register
   215            Bits32  ierr;           //! Internal Exception report register
   216            Context *excContext;//! Context structure filled by last exception
   217        };
   218    
   219        // -------- Module Constants --------
   220    
   221        /*
   222         *  Bitmasks for C64+ Exception Flag Register (EFR).
   223         */
   224        const Bits32 EFRSXF = 0x00000001;   //! Software exception flag
   225        const Bits32 EFRIXF = 0x00000002;   //! Internal exception flag
   226        const Bits32 EFREXF = 0x40000000;   //! EXCEP flag
   227        const Bits32 EFRNXF = 0x80000000;   //! NMI exception flag
   228    
   229        /*
   230         *  Bitmasks for C64+ Exception Clear Register (ECR).
   231         */
   232        const Bits32 ECRSXF = EFRSXF;       //! Software exception flag
   233        const Bits32 ECRIXF = EFRIXF;       //! Internal exception flag
   234        const Bits32 ECREXF = EFREXF;       //! EXCEP flag
   235        const Bits32 ECRNXF = EFRNXF;       //! NMI exception flag
   236    
   237        /*
   238         *  Bitmasks for C64+ Internal Exception Report Register (IERR).
   239         */
   240        const Bits32 IERRIFX = 0x00000001;  //! Instruction fetch exception
   241        const Bits32 IERRFPX = 0x00000002;  //! Fetch packet exception
   242        const Bits32 IERREPX = 0x00000004;  //! Execute packet exception
   243        const Bits32 IERROPX = 0x00000008;  //! Opcode exception
   244        const Bits32 IERRRCX = 0x00000010;  //! Resource conflict exception
   245        const Bits32 IERRRAX = 0x00000020;  //! Resource access exeption
   246        const Bits32 IERRPRX = 0x00000040;  //! Priviledge exception
   247        const Bits32 IERRLBX = 0x00000080;  //! Loop buffer exception
   248        const Bits32 IERRMSX = 0x00000100;  //! Missed stall exception
   249    
   250        /*
   251         *  Exception-related bitmasks for C64+ Task State Register (TSR).
   252         */
   253        const Bits32 TSRGEE = 0x00000004;   //! Global exception enable
   254        const Bits32 TSRXEN = 0x00000008;   //! External exception enable
   255        const Bits32 TSREXC = 0x00000400;   //! Exception processing
   256    
   257        /* TSR Privilege Mode bits */
   258        const Bits32 TSRCXM   = 0x000000C0; //! Current execution mode bits
   259        const Bits32 TSRCXMSHIFT = 6;       //! TSR.CXM left shift value
   260        const Bits32 TSRCXMSV = 0x00000000; //! Supervisor mode
   261        const Bits32 TSRCXMUS = 0x00000040; //! User mode

   263        /* size of buffer (in bytes) to hold exception context */

   274        };

   276        // -------- Module Parameters --------
   277    
   278        /*!
   279         *  If true, the exception context is saved to an internal buffer.
   280         *  If false, the exception context is saved to the bottom of the isr stack
   281         *  and no memory for the internal buffer is allocated.
   282         */
   283        config Bool useInternalBuffer = false;
   284    
   285        /*!
   286         *  enableExternalMPC - Enable handling of Memory Protection
   287         *  Controller (MPC) exceptions.
   288         */
   289        config Bool enableExternalMPC = false;
   290    
   291        /*!
   292         *  enablePrint - Enable print of exception details and Register values
   293         */
   294        config Bool enablePrint = true;
   295    
   296        /*!
   297         *  exceptionHook - Function hook called by handler
   298         *  This is called anytime an exception occurs.
   299         */
   300        config FuncPtr exceptionHook = null;
   301    
   302        /*!
   303         *  internalHook - Function hook called by internalHandler
   304         *  Function is only called when an internal exception has occurred.
   305         */
   306        config FuncPtr internalHook = null;
   307    
   308        /*!
   309         *  externalHook - Function hook called by externalHandler
   310         *  Function is only called when an external exception has occurred.
   311         */
   312        config FuncPtr externalHook = null;
   313    
   314        /*!
   315         *  nmiHook - Function hook called by nmiHandler
   316         *  Function is called for legacy NMI exceptions only
   317         */
   318        config FuncPtr nmiHook = null;
   319    
   320        /*! returnHook - Function hook called at the end of Exception_dispatch */
   321        config FuncPtr returnHook = null;
   322    
   323        /*!
   324         *  getLastStatus - Fills passed status structure with the Status
   325         *  fields that were recorded by the last invocation of
   326         *  dispatch(), handler() and internalHandler().
   327         *  The 'excContext' is valid only in the scope of sub-handler
   328         *  "Hook" functions.
   329         */
   330        @DirectCall
   331        Void getLastStatus(Status *status);
   332    
   333        /*!
   334         *  clearLastStatus - Clears internal Status structure.
   335         */
   336        @DirectCall
   337        Void clearLastStatus();
   338    
   339        /*!
   340         *  setReturnPtr - Configures dispatch() to "return" (branch) to the
   341         *  passed ptr.
   342         */
   343        @DirectCall
   344        FuncPtr setReturnPtr(FuncPtr ptr);
   345    
   346        /*!
   347         *  evtEvtClear - Clear a C64+ event from the EVTFLAG register.
   348         */
   349        @DirectCall
   350        Void evtEvtClear(UInt event);
   351    
   352        /*!
   353         *  evtExpMaskEnable - Enable a C64+ event to generate an exception.
   354         */
   355        @DirectCall
   356        Void evtExpMaskEnable(UInt event);
   357    
   358        /*! @_nodoc
   359         *  dispatch - The default low-level dispatcher, plugged into the
   360         *  C64+ NMI vector.
   361         */
   362        @DirectCall
   363        Void dispatch();
   364    
   365    internal:
   366    
   367        /* MPC CPU Access Memory Protecton Fault Event numbers */
   368        const UInt EVTPMCCMPA = 120;        /* PMC memory protection fault event */
   369        const UInt EVTDMCCMPA = 122;        /* DMC memory protection fault event */
   370        const UInt EVTUMCCMPA = 124;        /* UMC memory protection fault event */
   371        const UInt EVTEMCCMPA = 126;        /* EMC memory protection fault event */
   372    
   373        /*!
   374         *  handler - The high-level dispatcher, called by dispatch().
   375         *  Performs the following steps in order:
   376         *    a. records EFR/NRP/NTSR in a Status structure
   377         *    b. logs EFR/NRP/NTSR.CXM with text output to module's logger
   378         *    c. calls exceptionHook
   379         *    d. clears EFR
   380         *    e. calls into subhandlers
   381         *    f. aborts system
   382         */
   383        Void handler(Bool abortFlag);
   384    
   385        /*!
   386         *  internalHandler - Internal exception handler called by
   387         *  handler().  Performs the following steps in order:
   388         *    a. records IERR in a Status structure
   389         *    b. logs IERR with text output to module's logger
   390         *    c. calls internalHook
   391         *    d. clears IERR
   392         */
   393        Void internalHandler();
   394    
   395        /*!
   396         *  externalHandler - External exception handler called by
   397         *  handler().  Performs the following steps in order:
   398         *    a. logs text output to module's logger
   399         *    b. calls externalHook
   400         */
   401        Void externalHandler();
   402    
   403        /*!
   404         *  nmiHandler - Legacy NMI handler called by handler().
   405         *  Performs the following steps in order:
   406         *    a. logs text output to module's logger
   407         *    b. calls nmiHook
   408         */
   409        Void nmiHandler();
   410    
   411        /*
   412         *  ======== decodeMpfsr ========
   413         */
   414        Void decodeMpfsr(UInt mpfsr);
   415    
   416        struct Module_State {
   417            Bits32  efr;
   418            Bits32  nrp;
   419            Bits32  ntsr;
   420            Bits32  ierr;
   421            FuncPtr returnHook;
   422            Context *excContext;
   423            Char    scratch[16];    // sratch memory used by dispatch
   424            Char    *excPtr;        // points to isrStack or contextBuf
   425            Char    contextBuf[];   // Need at least 74 words for context buf
   426        };
   427    }
