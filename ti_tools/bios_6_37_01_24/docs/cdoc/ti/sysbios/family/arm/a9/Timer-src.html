    46    /*!
    47     *  ======== Timer ========
    48     *  Cortex A9 Timer Peripheral Manager
    49     *
    50     *  The Cortex A9 Timer Peripheral Manager utilizes the A9's internal Private
    51     *  Timer. The A9 Private Timer is clocked at half the CPU clock rate. The
    52     *  A9 Private Timer has 32 bits of period resolution and the interrupt
    53     *  generated by this timer is connected to private peripheral interrupt
    54     *  number 29 on the A9's generic interrupt controller, GIC
    55     *  (see {@link ti.sysbios.family.arm.gic.Hwi}).

    65     *    <!--                                                                                                                 -->

   107    @InstanceFinalize       /* To cleanup */
   108    @InstanceInitError      /* To report unavailability of timer */
   109    @ModuleStartup          /* to configure static timers */
   110    
   111    module Timer inherits ti.sysbios.interfaces.ITimer
   112    {
   113        /*! 
   114         *  Max value of Timer period for PeriodType_COUNTS
   115         */
   116        const UInt32 MAX_PERIOD = 0xFFFFFFFF;
   117    
   118        /*! Number of timer peripherals on chip */
   119        const Int NUM_TIMER_DEVICES = 1;
   120    
   121        /*! @_nodoc */
   122        @XmlDtd
   123        metaonly struct BasicView {
   124            Ptr         halTimerHandle;
   125            String      label;
   126            String      startMode;
   127            String      runMode;
   128            String      period;
   129            String      periodType;
   130            String      prescale;
   131            UInt        intNum;
   132            String      tickFxn[];
   133            UArg        arg;  
   134            String      extFreq;
   135            String      hwiHandle;
   136        };
   137    
   138        /*! @_nodoc */
   139        metaonly struct DeviceView {
   140            String      deviceAddr;
   141            UInt        intNum;
   142            String      period;
   143            String      currCount;
   144            String      remainingCount;
   145            String      prescale;
   146        }
   147    
   148        /*! @_nodoc */
   149        @Facet
   150        metaonly config ViewInfo.Instance rovViewInfo = 
   151            ViewInfo.create({
   152                viewMap: [
   153                [
   154                    'Basic',
   155                    {
   156                        type: ViewInfo.INSTANCE,
   157                        viewInitFxn: 'viewInitBasic',
   158                        structName: 'BasicView'
   159                    }
   160                ],
   161                [
   162                    'Device',
   163                    {
   164                        type: ViewInfo.INSTANCE,
   165                        viewInitFxn: 'viewInitDevice',
   166                        structName: 'DeviceView'
   167                    }
   168                ],
   169                ]
   170            });
   171    
   172        /*! 
   173         *  Error raised when timer id specified is not supported.
   174         */
   175        config Error.Id E_invalidTimer  = 
   176            {msg: "E_invalidTimer: Invalid Timer Id %d"}; 
   177    
   178        /*! 
   179         *  Error raised when timer requested is in use
   180         */
   181        config Error.Id E_notAvailable  = 
   182            {msg: "E_notAvailable: Timer not available %d"}; 
   183    
   184        /*! 
   185         *  Error raised when period requested is not supported
   186         */
   187        config Error.Id E_cannotSupport  = 
   188            {msg: "E_cannotSupport: Timer cannot support requested period %d"}; 
   189    
   190        /*!
   191         *  ======== intNumDef ========
   192         *  A9 Private Timer Interrupt number
   193         * 
   194         *  A9 Private Timer Interrupt is forwarded to GIC's private peripheral
   195         *  interrupt and has a architecture defined fixed mapping that should
   196         *  be same across all A9 variants.
   197         *
   198         *  Private Timer -> PPI1 or Interrupt #29
   199         */
   200        config UInt intNumDef;
   201    
   202        /*!
   203         *  @_nodoc
   204         *  ======== anyMask ========
   205         *  Available mask to be used when select = Timer_ANY
   206         */
   207        config UInt anyMask = 0x1;
   208    
   209        /*!
   210         *  @_nodoc
   211         *  ======== availMask ========
   212         *  Available mask tracks the available/free timer peripherals
   213         */
   214        config UInt availMask = 0x1;
   215    
   216        /*!
   217         *  @_nodoc
   218         *  Private timer registers. Symbol "Timer_deviceRegs" is a
   219         *  physical device.
   220         */
   221        struct DeviceRegs {
   222            UInt32 LOAD;
   223            UInt32 COUNTER;
   224            UInt32 CONTROL;
   225            UInt32 INTSTATUS;
   226        };
   227    
   228        extern volatile DeviceRegs deviceRegs;
   229    
   230        /*!
   231         *  @_nodoc
   232         *  ======== stub ========
   233         *  This stub is used to acknowledge a Timer interrupt.
   234         *
   235         *  @param(arg)     Timer Handle.
   236         */
   237        @DirectCall
   238        Void stub(UArg arg);
   239    
   240        /*!
   241         *  @_nodoc
   242         *  ======== getHandle ========
   243         *  Used by TimestampProvider module to get hold of timer handle used by 
   244         *  Clock.
   245         *
   246         *  @param(id)      timer Id.
   247         */
   248        @DirectCall
   249        Handle getHandle(UInt id);
   250    
   251    instance:
   252    
   253        /*! Hwi Params for Hwi Object. Default is null.*/
   254        config Hwi.Params *hwiParams = null;
   255    
   256        /*!
   257         *  ======== prescale ========
   258         *  Prescale factor.
   259         *
   260         *  The Prescale factor can be used to achieve longer timer periods.
   261         *  With a prescale factor specified, the actual timer period is
   262         *  period * (prescale + 1).
   263         */
   264        config UInt8 prescale = 0;
   265    
   266        /*!
   267         *  ======== reconfig ========
   268         *  Used to modify static timer instances at runtime.
   269         *
   270         *  @param(timerParams)     timer Params
   271         *  @param(tickFxn)         functions that runs when timer expires.
   272         */
   273        @DirectCall
   274        Void reconfig(FuncPtr tickFxn, const Params *timerParams, Error.Block *eb);
   275    
   276        /*!
   277         *  ======== setPrescale ========
   278         *  Set timer prescale value.
   279         *
   280         *  This function sets the value of the prescaler, and will also reload
   281         *  the timer counter and prescale registers.
   282         *
   283         *  @param(preScaler)       prescale value
   284         */
   285        @DirectCall
   286        Void setPrescale(UInt8 preScaler);
   287    
   288        /*!
   289         *  ======== getPrescale ========
   290         *  Get timer prescale value.
   291         *
   292         *  @b(returns)             prescale value
   293         */
   294        @DirectCall
   295        UInt8 getPrescale();
   296    
   297    internal:   /* not for client use */
   298       
   299        /*!
   300         *  ======== startupNeeded ========
   301         *  Flag used to prevent misc code from being brought in
   302         *  un-necessarily
   303         */
   304        config Bool startupNeeded = false;
   305    
   306        /*!
   307         *  ======== privTimerRegBaseAddress ========
   308         *  Cortex-A9 Private Timer Register base address
   309         */
   310        metaonly config Ptr privTimerRegBaseAddress;
   311    
   312        /*
   313         *  ======== initDevice ========
   314         *  reset timer to its resting state
   315         */
   316        Void initDevice(Object *timer);
   317    
   318        /*
   319         *  ======== postInit ========
   320         *  finish initializing static and dynamic Timers
   321         */
   322        Int postInit(Object *timer, Error.Block *eb);
   323    
   324        /*
   325         *  ======== checkOverflow ========
   326         */
   327        Bool checkOverflow(UInt32 a, UInt32 b);
   328    
   329        struct Instance_State {
   330            Bool                staticInst;
   331            ITimer.RunMode      runMode;    
   332            ITimer.StartMode    startMode;
   333            UInt32              period;
   334            ITimer.PeriodType   periodType;    
   335            UInt                intNum;
   336            UInt8               prescale;
   337            UArg                arg;  
   338            Hwi.FuncPtr         tickFxn;
   339            Types.FreqHz        extFreq;
   340            Hwi.Handle          hwi;
   341        }
   342    
   343        struct Module_State {
   344            UInt                availMask;  /* available peripherals */
   345            Handle              handle;     /* timer handle */
   346        }
   347    }
