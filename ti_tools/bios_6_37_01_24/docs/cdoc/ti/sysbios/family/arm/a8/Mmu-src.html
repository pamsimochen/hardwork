    44    /*!
    45     *  ======== Mmu ========
    46     *  Memory Management Unit Manager.
    47     *
    48     *  This module allows the ARM processor to map a virtual address to a
    49     *  different physical address and enable/disable the MMU.  It does this
    50     *  through a translation table in memory.  The translation table is
    51     *  16KB and manages only the first level descriptor table.  Each entry
    52     *  in the table defines the properties of memory areas of size 1MB.
    53     *  These properties include memory access permissions, cacheability,
    54     *  bufferability, and domain access.
    55     *
    56     *  By default, the MMU translation table is initialized with
    57     *  cache-enabled entries for every memory segment defined in the platform.
    58     *  Cache-disabled entries are also added for the peripheral
    59     *  addresses used by SYS/BIOS (i.e. Timers, Interrupt controller).
    60     *
    61     *  The translation table is placed in
    62     *  an output section called "ti.sysbios.family.arm.a8.mmuTableSection".
    63     *  This section is placed into the platform's default dataMemory and
    64     *  in order to minimize object file size, specified to not be initialized
    65     *  via the "NOLOAD" type on GNU compilers and "NOINIT" on TI compilers.
    66     *
    67     *  This module does not manage the second level descriptor tables.
    68     *  A 'SECTION' mapped access requires only a first level fetch.  In
    69     *  this case, there is no need for a second level descriptor table.
    70     *  A 'PAGE_TABLE' mapped access requires a second level
    71     *  descriptor table which can be supplied by the user.
    72     *
    73     *  The following is an example of how to place the MMU table
    74     *  and how to enable L1 data + L2 caching for the address range
    75     *  0x80000000-0x90000000 in the *.cfg file:
    76     *
    77     *  @p(code)
    78     *
    79     *    // For Cortex-A8
    80     *    var Cache = xdc.useModule('ti.sysbios.family.arm.a8.Cache');
    81     *    // For Cortex-A9
    82     *    var Cache = xdc.useModule('ti.sysbios.family.arm.a9.Cache');
    83     *
    84     *    var Mmu = xdc.useModule('ti.sysbios.family.arm.a8.Mmu');
    85     *
    86     *    // Enable the cache
    87     *    Cache.enableCache = true;
    88     *
    89     *    // Enable the MMU (Required for L1/L2 data caching)
    90     *    Mmu.enableMMU = true;
    91     *
    92     *    // descriptor attribute structure
    93     *    var attrs = {
    94     *        type: Mmu.FirstLevelDesc_SECTION,  // SECTION descriptor
    95     *        tex: 0x1,
    96     *        bufferable: true,                  // bufferable
    97     *        cacheable: true,                   // cacheable
    98     *    };
    99     *
   100     *    // Set the descriptor for each entry in the address range
   101     *    for (var i=0x80000000; i < 0x90000000; i = i + 0x00100000) {
   102     *        // Each 'SECTION' descriptor entry spans a 1MB address range
   103     *        Mmu.setFirstLevelDescMeta(i, i, attrs);
   104     *    }
   105     *
   106     *    var memmap = Program.cpu.memoryMap;
   107     *    var DDR = null;
   108     *
   109     *    // Find DDR in memory map
   110     *    for (var i=0; i < memmap.length; i++) {
   111     *        if (memmap[i].name == "DDR") {
   112     *            DDR = memmap[i];
   113     *        }
   114     *    }
   115     *
   116     *    // Place the MMU table in the DDR memory segment if it exists
   117     *    if (DDR != null) {
   118     *        var sectionName = "ti.sysbios.family.arm.a8.mmuTableSection";
   119     *        Program.sectMap[sectionName] = new Program.SectionSpec();
   120     *        Program.sectMap[sectionName].type = "NOLOAD"; // NOINIT for TI Tools
   121     *        Program.sectMap[sectionName].loadSegment = "DDR";
   122     *    }
   123     *    else {
   124     *        print("No DDR memory segment was found");
   125     *    }
   126     *
   127     *  @p
   128     *
   129     *  The following example demonstrates how to add a peripheral's address
   130     *  to the MMU table so that it can be accessed by code at runtime:
   131     *
   132     *  @p(code)
   133     *    // For Cortex-A8
   134     *    var Cache = xdc.useModule('ti.sysbios.family.arm.a8.Cache');
   135     *    // For Cortex-A9
   136     *    var Cache = xdc.useModule('ti.sysbios.family.arm.a9.Cache');
   137     *
   138     *    var Mmu = xdc.useModule('ti.sysbios.family.arm.a8.Mmu');
   139     *
   140     *    // Enable the cache
   141     *    Cache.enableCache = true;
   142     *
   143     *    // Enable the MMU (Required for L1/L2 data caching)
   144     *    Mmu.enableMMU = true;
   145     *
   146     *    // Force peripheral section to be NON cacheable strongly-ordered memory
   147     *    var peripheralAttrs = {
   148     *        type : Mmu.FirstLevelDesc_SECTION, // SECTION descriptor
   149     *        tex: 0,
   150     *        bufferable : false,                // bufferable
   151     *        cacheable  : false,                // cacheable
   152     *        shareable  : false,                // shareable
   153     *        noexecute  : true,                 // not executable
   154     *    };
   155     *
   156     *    // Define the base address of the 1 Meg page
   157     *    // the peripheral resides in.
   158     *    var peripheralBaseAddr = 0xa0400000;
   159     *
   160     *    // Configure the corresponding MMU page descriptor accordingly
   161     *    Mmu.setFirstLevelDescMeta(peripheralBaseAddr,
   162     *                              peripheralBaseAddr,
   163     *                              peripheralAttrs);
   164     *  @p
   165     *
   166     *  Notes:
   167     *  @p(blist)
   168     *      -The 'Supersection' table descriptor is NOT supported.
   169     *      -The 'not global' page descriptor setting is NOT supported.
   170     *      -The 'non-secure' page descriptor setting is NOT supported.
   171     *      -There are size and alignment requirements on the second
   172     *       level descriptor tables depending on the page size.
   173     *      -See the ARM Architecture Reference Manual for more info.
   174     *  @p
   175     *
   176     */
   177    
   178    @Template ("./Mmu.xdt") /* generate function to init MMU page table */
   179    @ModuleStartup          /* call to initTableBuf() in startup */
   180    
   181    module Mmu
   182    {
   183        // -------- ROV views --------
   184    
   185        /*! @_nodoc */
   186        metaonly struct PageView {
   187            String      Type;
   188            String      Tex;
   189            Ptr         AddrVirtual;
   190            Ptr         AddrPhysical;
   191            Ptr         Level2TablePtr;
   192            Bool        Bufferable;
   193            Bool        Cacheable;
   194            Bool        Shareable;
   195            Bool        Noexecute;
   196            String      L1DWritePolicy;
   197            UInt        IMP;
   198            UInt        Domain;
   199            String      Access;
   200        };
   201    
   202        @Facet
   203        metaonly config ViewInfo.Instance rovViewInfo =
   204            ViewInfo.create({
   205                viewMap: [
   206                    ['0x0-0x7FFFFFFF', {
   207                        type: ViewInfo.MODULE_DATA,
   208                        viewInitFxn: 'viewPages1',
   209                        structName: 'PageView'
   210                    }],
   211                    ['0x80000000-0x9FFFFFFF', {
   212                        type: ViewInfo.MODULE_DATA,
   213                        viewInitFxn: 'viewPages2',
   214                        structName: 'PageView'
   215                    }],
   216                    ['0xA0000000-0xBFFFFFFF', {
   217                        type: ViewInfo.MODULE_DATA,
   218                        viewInitFxn: 'viewPages3',
   219                        structName: 'PageView'
   220                    }],
   221                    ['0xC0000000-0xDFFFFFFF', {
   222                        type: ViewInfo.MODULE_DATA,
   223                        viewInitFxn: 'viewPages4',
   224                        structName: 'PageView'
   225                    }],
   226                    ['0xE0000000-0xFFFFFFFF', {
   227                        type: ViewInfo.MODULE_DATA,
   228                        viewInitFxn: 'viewPages5',
   229                        structName: 'PageView'
   230                    }]
   231               ]
   232           });
   233    
   234        /*!
   235         *  First Level descriptors
   236         *
   237         *  Different descriptor type encodings:
   238         *  0b00    ->  Invalid or Fault entry
   239         *  0b01    ->  Page table entry
   240         *  0b10    ->  Section descriptor
   241         *  0b11    ->  Reserved
   242         */
   243        enum FirstLevelDesc {
   244            FirstLevelDesc_FAULT = 0,      /*! Virtual address is unmapped     */
   245            FirstLevelDesc_PAGE_TABLE = 1, /*! Page table addr descriptor      */
   246            FirstLevelDesc_SECTION = 2,    /*! Section descriptor              */
   247            FirstLevelDesc_RESERVED = 3    /*! Reserved                        */
   248        };
   249    
   250        /*!
   251         *  Structure for setting first level descriptor entries
   252         *
   253         *  The B (Bufferable), C (Cacheable), and TEX (Type extension) bits
   254         *  in the translation table descriptor define the memory region
   255         *  attributes.
   256         *
   257         *  See the 'Short-descriptor translation table format' section of the
   258         *  ARM v7-AR Architecture Reference Manual issue C for more details.
   259         */
   260        struct FirstLevelDescAttrs {
   261            FirstLevelDesc type;    /*! first level descriptor type             */
   262            Bool  bufferable;       /*! is memory section bufferable            */
   263            Bool  cacheable;        /*! is memory section cacheable             */
   264            Bool  shareable;        /*! is memory section shareable             */
   265            Bool  noexecute;        /*! is memory section not executable        */
   266            UInt8 imp;              /*! implementation defined                  */
   267            UInt8 domain;           /*! domain access control value 0-15        */
   268            UInt8 accPerm;          /*! access permission bits value 0-7        */
   269            UInt8 tex;              /*! memory region attr type extension field */
   270        };
   271    
   272        // Asserts
   273    
   274        /*!
   275         *  ======== A_nullPointer ========
   276         *  Assert raised when a pointer is null
   277         */
   278        config Assert.Id A_nullPointer  = {
   279            msg: "A_nullPointer: Pointer is null"
   280        };
   281    
   282        /*!
   283         *  ======== A_unknownDescType ========
   284         *  Assert raised when the descriptor type is not recognized.
   285         */
   286        config Assert.Id A_unknownDescType = {
   287            msg: "A_unknownDescType: Descriptor type is not recognized"
   288        };
   289    
   290        /*! default descriptor attributes structure */
   291        config FirstLevelDescAttrs defaultAttrs = {
   292            type: FirstLevelDesc_SECTION,   /* SECTION descriptor */
   293            bufferable: false,              /* false by default     */
   294            cacheable: false,               /* false by default     */
   295            shareable: false,               /* false by default */
   296            noexecute: false,               /* false by default     */
   297            imp: 1,                         /* set to 1 for A8/A9 devices */
   298            domain: 0,                      /* default Domain is 0 */
   299            accPerm: 3,                     /* allow read/write */
   300            tex: 1                          /* 1 by default */
   301        };
   302    
   303        /*!
   304         *  ======== enableMMU ========
   305         *  Configuration parameter to enable MMU.
   306         */
   307        config Bool enableMMU = true;
   308    
   309        /*!
   310         *  ======== cachePlatformMemory ========
   311         *  Flag to automatically mark platform's code/data/stack memory as
   312         *  cacheable in MMU descriptor table
   313         *
   314         *  By default, all memory regions defined in the platform an
   315         *  application is built with are marked as cacheable.
   316         *
   317         *  @see xdc.bld.Program#platform
   318         *
   319         *  If manual configuration of memory regions is required, set
   320         *  this config parameter to 'false'.
   321         */
   322        metaonly config Bool cachePlatformMemory = true;
   323    
   324        /*!
   325         *  ======== setFirstLevelDescMeta ========
   326         *  Statically sets the descriptor for the virtual address.
   327         *
   328         *  The first level table entry for the virtual address is mapped
   329         *  to the physical address with the attributes specified. The
   330         *  descriptor table is effective when the MMU is enabled.
   331         *
   332         *  @param(virtualAddr)  The modified virtual address
   333         *  @param(phyAddr)      The physical address
   334         *  @param(attrs)        Pointer to first level descriptor attribute struct
   335         */
   336        metaonly Void setFirstLevelDescMeta(Ptr virtualAddr, Ptr phyAddr,
   337                                            FirstLevelDescAttrs attrs);
   338    
   339        /*!
   340         *  ======== disable ========
   341         *  Disables the MMU.
   342         *
   343         *  If the MMU is already disabled, then simply return.
   344         *  Otherwise this function does the following:
   345         *  If the L1 data cache is enabled, write back invalidate all
   346         *  of L1 data cache.  If the L1 program cache is enabled,
   347         *  invalidate all of L1 program cache. This function does not
   348         *  change the cache L1 data/program settings.
   349         */
   350        @DirectCall
   351        Void disable();
   352    
   353        /*!
   354         *  ======== enable ========
   355         *  Enables the MMU.
   356         *
   357         *  If the MMU is already enabled, then simply return.
   358         *  Otherwise this function does the following:
   359         *  If the L1 program cache is enabled, invalidate all of L1
   360         *  program cache.  This function does not change the L1
   361         *  data/program cache settings.
   362         */
   363        @DirectCall
   364        Void enable();
   365    
   366        /*!
   367         *  ======== initDescAttrs() ========
   368         *  Initializes the first level descriptor attribute structure
   369         *
   370         *  @param(attrs)      Pointer to first level descriptor attribute struct
   371         */
   372        @DirectCall
   373        Void initDescAttrs(FirstLevelDescAttrs *attrs);
   374    
   375        /*!
   376         *  ======== isEnabled ========
   377         *  Determines if the MMU is enabled
   378         */
   379        @DirectCall
   380        Bool isEnabled();
   381    
   382        /*!
   383         *  ======== setFirstLevelDesc ========
   384         *  Sets the descriptor for the virtual address.
   385         *
   386         *  The first level table entry for the virtual address is mapped
   387         *  to the physical address with the attributes specified. The
   388         *  descriptor table is effective when the MMU is enabled.
   389         *
   390         *  @param(virtualAddr)  The modified virtual address
   391         *  @param(phyAddr)      The physical address
   392         *  @param(attrs)        Pointer to first level descriptor attribute struct
   393         */
   394        @DirectCall
   395        Void setFirstLevelDesc(Ptr virtualAddr, Ptr phyAddr,
   396                               FirstLevelDescAttrs *attrs);
   397    
   398        /*!
   399         *  ======== getPhysicalAddr ========
   400         *  Returns the translated physical address for the given virtual address.
   401         *  If the virtual address cannot be translated i.e. the virtual address
   402         *  would generate a fault if translated, ~(0) is returned.
   403         *
   404         *  @param(virtualAddr)  virtual address
   405         *  @b(returns)          translated physical address
   406         */
   407        @DirectCall
   408        Ptr getPhysicalAddr(Ptr virtualAddr);
   409    
   410    internal:
   411    
   412        /*! static array to hold first level dscriptor table */
   413        metaonly config UInt32 tableBuf[];
   414    
   415        /*!
   416         *  ======== init ========
   417         *  initialize mmu registers
   418         */
   419        Void init();
   420    
   421        /*!
   422         *  ======== enableAsm ========
   423         *  Assembly function to enable the MMU.
   424         */
   425        Void enableAsm();
   426    
   427        /*!
   428         *  ======== disableAsm ========
   429         *  Assembly function to disable the MMU.
   430         */
   431        Void disableAsm();
   432    
   433        /*! function generated to initialize first level descriptor table */
   434        Void initTableBuf(UInt32 *mmuTableBuf);
   435    
   436        /*!
   437         *  ======== getPhysicalAddrI ========
   438         *  Returns the translated physical address for the given virtual address.
   439         *  If the virtual address cannot be translated i.e. the virtual address
   440         *  would generate a fault if translated, ~(0) is returned.
   441         */
   442        Ptr getPhysicalAddrI(Ptr virtualAddr);
   443    
   444        /*! Module state */
   445        struct Module_State {
   446            UInt32 tableBuf[];      /*! 16KB array for first-level descriptors */
   447        }
   448    }
