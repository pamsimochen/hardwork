    37    package ti.sysbios.family.arm.exc;
    38    
    39    import xdc.rov.ViewInfo;
    40    
    41    import xdc.runtime.Error;
    42    
    43    import ti.sysbios.BIOS;
    44    
    45    /*!
    46     *  ======== Exception ========
    47     *  Exception Module
    48     *
    49     *  The Exception module is a basic ARM exception handler.
    50     *  When an exception occurs, the execution state is saved into
    51     *  a buffer with a structure of type {@link #ExcContext ExcContext}.
    52     *
    53     *  By default, a formatted dump of this structure is also sent to
    54     *  the console using System_printf(). This behavior can be suppressed
    55     *  by setting the {@link #enableDecode} config parameter to 'false'.
    56     *
    57     *  An ROV view of the exception context is provided when
    58     *  attached to the target board with CCS.
    59     *
    60     *  @a(Notes)
    61     *  No attempt is made to return from an exception as these are
    62     *  generally fatal in nature.
    63     *
    64     *  When used, the Exception module will override the default
    65     *  exception handlers provided by the current Hwi module.
    66     */
    67    
    68    @ModuleStartup          /* generates call to Exception_Module_startup at startup */
    69    module Exception
    70    {
    71        /*! Exception hook function type definition. */
    72        typedef Void (*ExceptionHookFuncPtr)(ExcContext *);
    73    
    74        /* Exception types */
    75        enum Type {
    76            Type_Supervisor = 0x13,         /*! Supervisor (SWI). */
    77            Type_PreAbort   = 0x17,         /*! Abort exception. */
    78            Type_DataAbort  = 0x18,         /*! Abort exception. */
    79            Type_UndefInst  = 0x1b          /*! Undefined instruction exception. */
    80        };
    81    
    82        /*!
    83         *  Exception Context - Register contents at the time of an exception.
    84         */
    85        struct ExcContext {
    86            /* Thread Context */
    87            BIOS.ThreadType threadType; /* Type of thread executing at */
    88                                        /* the time the exception occurred */
    89            Ptr     threadHandle;       /* Handle to thread executing at */
    90                                        /* the time the exception occurred */
    91            Ptr     threadStack;        /* Address of stack contents of thread */
    92                                        /* executing at the time the exception */
    93                                        /* occurred */
    94            SizeT   threadStackSize;    /* size of thread stack */
    95    
    96            Type    type;
    97    
    98            /* Internal Registers */
    99            Ptr     r0;
   100            Ptr     r1;
   101            Ptr     r2;
   102            Ptr     r3;
   103            Ptr     r4;
   104            Ptr     r5;
   105            Ptr     r6;
   106            Ptr     r7;
   107            Ptr     r8;
   108            Ptr     r9;
   109            Ptr     r10;
   110            Ptr     r11;
   111            Ptr     r12;
   112            Ptr     sp;
   113            Ptr     lr;
   114            Ptr     pc;
   115            Ptr     psr;
   116    
   117            /* Fault registers */
   118            Ptr     dfsr;
   119            Ptr     ifsr;
   120            Ptr     dfar;
   121            Ptr     ifar;
   122        }
   123    
   124        @Facet
   125        metaonly config ViewInfo.Instance rovViewInfo =
   126            ViewInfo.create({
   127                viewMap: [
   128                    ['Basic',
   129                        {
   130                            type: ViewInfo.TREE,
   131                            viewInitFxn: 'viewInitBasic',
   132                            structName: 'ExcContext'
   133                        }
   134                    ]
   135                ]
   136            });
   137    
   138        /*! Error raised when a SWI exception occurs */
   139        config Error.Id E_swi = {
   140            msg: "E_swi: pc = 0x%08x, lr = 0x%08x."
   141        };
   142    
   143        /*! Error raised when an prefetch abort exception occurs */
   144        config Error.Id E_prefetchAbort = {
   145            msg: "E_prefetchAbort: pc = 0x%08x, lr = 0x%08x."
   146        };
   147    
   148        /*! Error raised when an data abort exception occurs */
   149        config Error.Id E_dataAbort = {
   150            msg: "E_dataAbort: pc = 0x%08x, lr = 0x%08x."
   151        };
   152    
   153        /*! Error raised when an undefined instruction exception occurs */
   154        config Error.Id E_undefinedInstruction = {
   155            msg: "E_undefinedInstruction: pc = 0x%08x, lr = 0x%08x."
   156        };
   157    
   158        /*!
   159         *  Enable full exception decoding, default is true.
   160         *
   161         *  When enabled, the exception handler will fully
   162         *  decode an exception and dump the registers to the
   163         *  system console.
   164         *
   165         *  When set to false, only an Error is printed on the console.
   166         *
   167         *  In either case, the full exception context is always
   168         *  saved and visible with ROV.
   169         */
   170        config Bool enableDecode = true;
   171    
   172        /*! @_nodoc
   173         *  Exception stack pointer. Default = null.
   174         *  (Indicates that stack is to be created using
   175         *  staticPlace()
   176         */
   177        config Ptr excStack = null;
   178    
   179        /*!
   180         * Exception stack size in MAUs.
   181         * Default is 128 bytes.
   182         */
   183        metaonly config SizeT excStackSize = 128;
   184    
   185        /*!
   186         *  Memory section used for Exception stack.
   187         *  Default is null, which results in the stack
   188         *  being placed in .bss.
   189         */
   190        metaonly config String excStackSection = null;
   191    
   192        /*!
   193         *  User Exception Context Buffer Address
   194         *
   195         *  By default, when an exception occurs, an {@link #ExcContext}
   196         *  structure is allocated on the exception stack and filled in
   197         *  by the exception handler.
   198         *
   199         *  If {@link #excContextBuffer} is initialized by the user, the
   200         *  {@link #ExcContext} structure will be placed at that address instead.
   201         *
   202         *  The buffer must be large enough to contain an {@link #ExcContext}
   203         *  structure.
   204         */
   205        metaonly config Ptr excContextBuffer;
   206    
   207        /*!
   208         *  User Exception Stack Buffer Address
   209         *
   210         *  When an exception occurs, a pointer to the base address
   211         *  of the stack being used by the thread causing the exception is stored
   212         *  in the ExcContext buffer.
   213         *
   214         *  If {@link #excStackBuffer} is initialized by the user, the
   215         *  entire contents of that stack will also be
   216         *  copied to the address specified.
   217         *
   218         *  The buffer must be large enough to contain the largest task stack or ISR
   219         *  stack defined in the application.
   220         */
   221        metaonly config Ptr excStackBuffer;
   222    
   223        /*!
   224         *  User Exception hook function.
   225         *
   226         *  Called just after exception context has been initialized.
   227         */
   228        config ExceptionHookFuncPtr excHookFunc = null;
   229    
   230    internal:   /* not for client use */
   231    
   232        /* assembly code mode registers setup */
   233        Void init();
   234    
   235        /*!
   236         *  ======== excHandlerAsm ========
   237         *  asm code common exception handler
   238         */
   239        Void excHandlerAsm();
   240    
   241        /*!
   242         *  ======== excHandlerDataAsm ========
   243         *  asm code Data Abort exception handler
   244         */
   245        Void excHandlerDataAsm();
   246    
   247        /*!
   248         *  ======== excHandler ========
   249         *  C function exception Handler
   250         */
   251        Void excHandler(UInt *excStack, UInt lr);
   252    
   253        /*!
   254         *  ======== excDumpContext ========
   255         */
   256        Void excDumpContext(UInt pc);
   257    
   258        struct Module_State {
   259            Bool        excActive;          // TRUE if an exception has occurred
   260            ExcContext  *excContext;        // Exception context
   261            Ptr         excStackBuffer;     // where to copy thread's stack
   262            Char        excStack[];         // buffer used for exception stack
   263            SizeT       excStackSize;
   264        };
   265    }
