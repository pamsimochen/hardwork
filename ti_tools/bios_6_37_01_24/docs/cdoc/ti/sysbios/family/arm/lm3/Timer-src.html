   102    @InstanceInitError      /* To report unavailability of timer */
   103    @ModuleStartup          /* to configure static timers */
   104    
   105    module Timer inherits ti.sysbios.interfaces.ITimer
   106    {
   107        /*! Max value of Timer period for PeriodType_COUNTS*/
   108        const UInt MAX_PERIOD = 0xffffffff;
   109    
   110        /*! Timer clock divider wrt cpu clock */
   111        const Int TIMER_CLOCK_DIVIDER = 1;
   112    
   113        /*! @_nodoc
   114         *  Min instructions to use in trigger().
   115         */
   116        const Int MIN_SWEEP_PERIOD = 1;
   117    
   118        /*! @_nodoc */
   119        @XmlDtd
   120        metaonly struct BasicView {
   121            Ptr         halTimerHandle;
   122            String      label;
   123            UInt        id;
   124            String      device;
   125            String      startMode;
   126            String      runMode;
   127            UInt        period;
   128            String      periodType;
   129            UInt        intNum;
   130            String      tickFxn[];
   131            UArg        arg;
   132            String      extFreq;
   133            String      hwiHandle;
   134        };
   135    
   136    
   137        /*! @_nodoc */
   138        metaonly struct ModuleView {
   139            String      availMask;      /* available 32-bit timer halves */
   140        }
   141    
   142        /*! @_nodoc */
   143        metaonly struct DeviceView {
   144            UInt        id;
   145            String      device;
   146            String      devAddr;
   147            UInt        intNum;
   148            String      runMode;
   149            UInt        period;
   150            UInt        currCount;
   151            UInt        remainingCount;
   152            String      state;
   153        };
   154    
   155        /*! @_nodoc */
   156        @Facet
   157        metaonly config ViewInfo.Instance rovViewInfo =
   158            ViewInfo.create({
   159                viewMap: [
   160                [
   161                    'Basic',
   162                    {
   163                        type: ViewInfo.INSTANCE,
   164                        viewInitFxn: 'viewInitBasic',
   165                        structName: 'BasicView'
   166                    }
   167                ],
   168                [
   169                    'Device',
   170                    {
   171                        type: ViewInfo.INSTANCE,
   172                        viewInitFxn: 'viewInitDevice',
   173                        structName: 'DeviceView'
   174                    }
   175                ],
   176                [
   177                    'Module',
   178                    {
   179                        type: ViewInfo.MODULE,
   180                        viewInitFxn: 'viewInitModule',
   181                        structName: 'ModuleView'
   182                    }
   183                ],
   184                ]
   185            });
   186    
   187        /*! Timer enable function type definition. */
   188        typedef Void (*TimerEnableFuncPtr)(Int);
   189    
   190        /*! Timer disable function type definition. */
   191        typedef Void (*TimerDisableFuncPtr)(Int);
   192    
   193        /*!
   194         *  Error raised when timer id specified is not supported.
   195         */
   196        config Error.Id E_invalidTimer  = {msg: "E_invalidTimer: Invalid Timer Id %d"};
   197    
   198        /*!
   199         *  Error raised when timer requested is in use
   200         */
   201        config Error.Id E_notAvailable  =
   202            {msg: "E_notAvailable: Timer not available %d"};
   203    
   204        /*!
   205         *  Error raised when period requested is not supported
   206         */
   207        config Error.Id E_cannotSupport  =
   208            {msg: "E_cannotSupport: Timer cannot support requested period %d"};
   209    
   210        /*!
   211         *  ======== anyMask ========
   212         *  Available mask to be used when select = Timer_ANY
   213         */
   214        config UInt anyMask = 0x3;
   215    
   216        /*!
   217         *  ======== enableFunc ========
   218         */
   219        config TimerEnableFuncPtr enableFunc = null;
   220    
   221        /*!
   222         *  ======== disableFunc ========
   223         */
   224        config TimerDisableFuncPtr disableFunc = null;
   225    
   226        /*!
   227         *  ======== getHandle ========
   228         *  Get Handle associated with a timer id.
   229         *
   230         *  @param(id)      timer Id.
   231         *  @b(returns)     timer Handle
   232         */
   233        @DirectCall
   234        Handle getHandle(UInt id);
   235    
   236        /*!
   237         *  ======== isrStub ========
   238         *  @_nodoc
   239         *  Timer interrupts must be acknowledged
   240         *
   241         *  @param(arg)     Timer object.
   242         */
   243        @DirectCall
   244        Void isrStub(UArg arg);
   245    
   246    instance:
   247    
   248        /*! Hwi Params for Hwi Object. Default is null. */
   249        config Hwi.Params *hwiParams = null;
   250    
   251        /*!
   252         *  ======== reconfig ========
   253         *  Used to modify static timer instances at runtime.
   254         *
   255         *  @param(timerParams)     timer Params
   256         *  @param(tickFxn)         function that runs when timer expires
   257         */
   258        @DirectCall
   259        Void reconfig(FuncPtr tickFxn, const Params *timerParams, Error.Block *eb);
   260    
   261    
   262    internal:   /* not for client use */
   263    
   264        /*
   265         *  ======== noStartupNeeded ========
   266         *  Flag used to prevent misc code from being brought in
   267         *  un-necessarily
   268         */
   269        config UInt startupNeeded = false;
   270    
   271        /*! Information about timer */
   272        struct TimerDevice {
   273            UInt intNum;
   274            Ptr  baseAddr;
   275        };
   276    
   277        /*!
   278         *  ======== numTimerDevices ========
   279         *  The number of logical timers on a device.
   280         */
   281        config Int numTimerDevices;
   282    
   283        /*!
   284         *  ======== WRALLOW ========
   285         *  Write protection register for critical registers;
   286         */
   287        config Ptr WRALLOW = null;
   288    
   289        /*!
   290         *  ======== disableStellaris ========
   291         */
   292        Void disableStellaris(Int id);
   293    
   294        /*!
   295         *  ======== enableStellaris ========
   296         */
   297        Void enableStellaris(Int id);
   298    
   299        /*
   300         *  ======== initDevice ========
   301         *  reset timer to its resting state
   302         */
   303        Void initDevice(Object *timer);
   304    
   305        /*
   306         *  ======== postInit ========
   307         *  finish initializing static and dynamic Timers
   308         */
   309        Int postInit(Object *timer, Error.Block *eb);
   310    
   311        /*
   312         *  ======== enableTimers ========
   313         *  enable timer register access
   314         *  called as a Startup.firstFxnw
   315         */
   316        Void enableTimers();
   317    
   318        struct Instance_State {
   319            Bool            staticInst;
   320            Int             id;
   321            RunMode         runMode;
   322            StartMode       startMode;
   323            UInt            period;
   324            PeriodType      periodType;
   325            UInt            intNum;
   326            UArg            arg;
   327            Hwi.FuncPtr     tickFxn;
   328            Types.FreqHz    extFreq;        /* external frequency in Hz */
   329            Hwi.Handle      hwi;
   330        }
   331    
   332        struct Module_State {
   333            UInt            availMask;      /* available peripherals */
   334            TimerDevice     device[];       /* timer device information */
   335            Handle          handles[];      /* array of handles based on id */
   336        }
   337    }
