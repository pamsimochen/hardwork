    38    import xdc.rov.ViewInfo;

    40    /*!
    41     *  ======== TimerSupport ========
    42     *  Vayu Timer Support.
    43     *
    44     *  The Vayu Timer Support module works in conjunction with the DMTimer
    45     *  module to manage the usage of the DMTimers by the various Vayu cores.
    46     *
    47     *  This module provides a timer available mask. The application can modify
    48     *  this available mask in the .cfg script to choose a different set of timers.
    49     *
    50     *  The default value of the timer {@link #availMask} depends on the target
    51     *  the application is being built for. The defaults are shown below:
    52     *
    53     *  @p(code)
    54     *   --------------------------------------------
    55     *  | Target       | Default availMask           |
    56     *   --------------------------------------------
    57     *  | A15          | 0x0202 (GPTimer 2 & 10)     |
    58     *  | DSP          | 0x0030 (GPTimer 5 & 6)      |
    59     *  | IPU(Dual-M4) | 0x050C (GPTimer 3,4,9 & 11) |
    60     *   --------------------------------------------
    61     *  @p
    62     *
    63     *  @a(Note)
    64     *  If the {@link #availMask} is changed, the interrupt sources
    65     *  corresponding to the new Timer Ids may need to be mapped in
    66     *  the interrupt crossbar module if they are not mapped by default.
    67     *
    68     *  See {@link ../../../timers/dmtimer/doc-files/TimerTables.html Timer Mapping Tables}
    69     *  to determine the default interruptNumber/eventId assigned to
    70     *  each DM Timer. In the Timer Mapping Table, an interrupt number
    71     *  of -1 implies the timer's interrupt source is not connected to
    72     *  the target cpu's interrupt controller. If you need to use a
    73     *  timer with an interrupt number of -1, you will need to first select
    74     *  an unused interrupt to use for the timer and then map the
    75     *  timer's interrupt source to the target cpu's interrupt controller
    76     *  using the
    77     *  {@link ti.sysbios.family.shared.vayu.IntXbar Interrupt Crossbar module}.
    78     *
    79     *  Below is an example showing how to statically override the default
    80     *  timer available mask in a configuration script for the DSP:
    81     *
    82     *  @p(code)
    83     *  var Timer = xdc.useModule('ti.sysbios.timers.dmtimer.Timer');
    84     *  var IntXbar = xdc.useModule('ti.sysbios.family.shared.vayu.IntXbar');
    85     *  var TimerSupport = xdc.useModule('ti.sysbios.family.shared.vayu.TimerSupport');
    86     *
    87     *  // The new availability mask includes GP Timers 13, 14, 15 & 16.
    88     *  // The interrupt sources for these timers is not connected to the
    89     *  // DSP and need to be connected by the user.
    90     *  TimerSupport.availMask = 0xF000;
    91     *
    92     *  // Connect GP Timer 13 to EventId 76 (XBAR Instance 45)
    93     *  IntXbar.connectMeta(45, 339);
    94     *
    95     *  // Connect GP Timer 14 to EventId 77 (XBAR Instance 46)
    96     *  IntXbar.connectMeta(46, 340);
    97     *
    98     *  // Connect GP Timer 15 to EventId 78 (XBAR Instance 47)
    99     *  IntXbar.connectMeta(47, 341);
   100     *
   101     *  // Connect GP Timer 16 to EventId 81 (XBAR Instance 49)
   102     *  IntXbar.connectMeta(49, 342);
   103     *
   104     *  // The DMTimer's timer mapping table shows that the intNum
   105     *  // for GP Timer 13, 14, 15 & 16 is -1 implying the intNum
   106     *  // has not been assigned by default. The below code utilizes
   107     *  // the DMTimer module's timerSettings array to assign
   108     *  // a default intNum to each of these timers.
   109     *  Timer.timerSettings[12].intNum = 14;
   110     *  Timer.timerSettings[13].intNum = 15;
   111     *  Timer.timerSettings[14].intNum = 4;
   112     *  Timer.timerSettings[15].intNum = 5;
   113     *
   114     *  var TimerParams = new Timer.Params();
   115     *  TimerParams.period = 5000;
   116     *  TimerParams.runMode = Timer.RunMode_ONESHOT;
   117     *  TimerParams.periodType = Timer.PeriodType_MICROSECS;
   118     *
   119     *  TimerParams.arg = 1;
   120     *  Timer.create(Timer.ANY, '&myIsr1', TimerParams);
   121     *
   122     *  TimerParams.arg = 2;
   123     *  Timer.create(Timer.ANY, '&myIsr2', TimerParams);
   124     *  @p
   125     *
   126     *  Below is an example showing how to statically override the default
   127     *  timer available mask in a configuration script for the A15:
   128     *
   129     *  @p(code)
   130     *  var Timer = xdc.useModule('ti.sysbios.hal.Timer');
   131     *  var DMTimer = xdc.useModule('ti.sysbios.timers.dmtimer.Timer');
   132     *
   133     *  // Override the default hal timer proxy for the A15 target
   134     *  Timer.TimerProxy = DMTimer;
   135     *
   136     *  var TimerSupport = xdc.module('ti.sysbios.family.shared.vayu.TimerSupport');
   137     *  TimerSupport.availMask = 0x000F;
   138     *
   139     *  var TimerParams = new Timer.Params();
   140     *  TimerParams.arg = 1;
   141     *  TimerParams.period = 5000;
   142     *  TimerParams.runMode = Timer.RunMode_ONESHOT;
   143     *  TimerParams.periodType = Timer.PeriodType_MICROSECS;
   144     *
   145     *  Timer.create(Timer.ANY, '&myIsr', TimerParams);
   146     *  @p
   147     *
   148     */
   149    
   150    @ModuleStartup
   151    
   152    module TimerSupport inherits ti.sysbios.interfaces.ITimerSupport
   153    {
   154        /*!
   155         *  ======== ModuleView ========
   156         *  @_nodoc
   157         */
   158        metaonly struct ModuleView {
   159            String availMask;
   160        }
   161    
   162        /*!
   163         *  ======== rovViewInfo ========
   164         *  @_nodoc
   165         */
   166        @Facet
   167        metaonly config ViewInfo.Instance rovViewInfo = 
   168            ViewInfo.create({
   169                viewMap: [
   170                [
   171                    'Module',
   172                    {
   173                        type: ViewInfo.MODULE,
   174                        viewInitFxn: 'viewInitModule',
   175                        structName: 'ModuleView'
   176                    }
   177                ],
   178                ]
   179            });
   180    
   181        /*!
   182         *  ======== availMask ========
   183         *  Default available mask for dmtimers
   184         *
   185         *  Used to set Timer module's availMask.
   186         *  The default value of the timer availMask depends on the target
   187         *  the application is being built for. The defaults are shown below:
   188         *
   189         *  @p(code)
   190         *   --------------------------------------------
   191         *  | Target       | Default availMask           |
   192         *   --------------------------------------------
   193         *  | A15          | 0x0202 (GPTimer 2 & 10)     |
   194         *  | DSP          | 0x0030 (GPTimer 5 & 6)      |
   195         *  | IPU(Dual-M4) | 0x050C (GPTimer 3,4,9 & 11) |
   196         *   --------------------------------------------
   197         *  @p
   198         */
   199         config Bits32 availMask = 0x0;
   200    
   201    internal:
   202    
   203        /*
   204         *  ======== getAnyMask ========
   205         *  Return the anyMask for this device.
   206         */
   207        metaonly Bits32 getAnyMask();
   208    
   209        /*
   210         *  ======== getAvailMask ========
   211         *  Return the availMask for this device.
   212         */
   213        metaonly Bits32 getAvailMask();
   214    }
