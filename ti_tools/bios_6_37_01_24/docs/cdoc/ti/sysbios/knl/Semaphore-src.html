    49    /*!
    50     *  ======== Semaphore ========
    51     *  Semaphore Manager
    52     *
    53     *  The Semaphore manager makes available a set of functions that manipulate
    54     *  semaphore objects. Semaphores can be used for task synchronization and
    55     *  mutual exclusion.
    56     *
    57     *  Semaphores can be counting semaphores or binary semaphores. Counting
    58     *  semaphores keep track of the number of times the semaphore has been posted
    59     *  with post(). This is useful, for example, if you have a group of resources
    60     *  that are shared between tasks. Such tasks might call pend() to see if a
    61     *  resource is available before using one.
    62     *
    63     *  Binary semaphores can have only two states: available (count = 1) and
    64     *  unavailable (count = 0). They can be used to share a single resource
    65     *  between
    66     *  tasks. They can also be used for a basic signaling mechanism, where the
    67     *  semaphore can be posted multiple times. Binary semaphores do not keep track
    68     *  of the count; they simply track whether the semaphore has been posted
    69     *  or not.
    70     *
    71     *  The Mailbox module uses a counting semaphore internally to manage the
    72     *  count of free (or full) mailbox elements. Another example of a counting
    73     *  semaphore is an ISR that might fill multiple buffers of data for
    74     *  consumption by a task. After filling each buffer, the ISR puts the buffer on
    75     *  a queue and calls post(). The task waiting for the data calls pend(), which
    76     *  simply decrements the semaphore count and returns or blocks if the count is
    77     *  0. The semaphore count thus tracks the number of full buffers available for
    78     *  the task.
    79     *
    80     *  pend() is used to wait for a semaphore. The timeout parameter allows the
    81     *  task to wait until a timeout, wait indefinitely, or not wait at all. The
    82     *  return value is used to indicate if the semaphore was signaled successfully.
    83     *
    84     *  post() is used to signal a semaphore. If a task is waiting for the
    85     *  semaphore, post() removes the task from the semaphore queue and puts it on
    86     *  the ready queue. If no tasks are waiting, post() simply increments the
    87     *  semaphore count and returns.  For a binary semaphore the count is always
    88     *  set to one.
    89     *
    90     *  @p(html)
    91     *  <h3> Calling Context </h3>

    97     *    <th>  Task  </th><th>  Main  </th><th>  Startup  </th></tr>
    98     *    <!--                                                                  -->
    99     *    <tr><td> {@link #Params_init}   </td><td>   Y    </td><td>   Y    </td>
   100     *    <td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>

   103     *    <tr><td> {@link #create}        </td><td>   N    </td><td>   N    </td>
   104     *    <td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   105     *    <tr><td> {@link #delete}        </td><td>   N    </td><td>   N    </td>

   109     *    <tr><td> {@link #getCount}      </td><td>   Y    </td><td>   Y    </td>
   110     *    <td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>

   151        /*!
   152         *  Semaphore types.
   153         *
   154         *  These enumerations specify the type of semaphore.
   155         *
   156         *  Tasks wait for the semaphore in FIFO order unless the PRIORITY
   157         *  option is chosen.
   158         *
   159         *  For PRIORITY semaphores, the pending task will be
   160         *  inserted in the waiting list before the first task that has
   161         *  lower priority.  This ensures that tasks of equal priority will
   162         *  pend in FIFO order.
   163         *
   164         *  @a(WARNING)
   165         *  PRIORITY semaphores have a potential to increase the interrupt
   166         *  latency in the system since interrupts are disabled while the list of
   167         *  tasks waiting on the  semaphore is scanned for the proper insertion
   168         *  point.  This is typically about a dozen instructions per waiting task.
   169         *  For example, if you have 10 tasks of higher priority waiting, then all
   170         *  10 will be checked with interrupts disabled before the new task is
   171         *  entered onto the list.
   172         */
   173        enum Mode {
   174            Mode_COUNTING = 0x0,           /*! Counting (FIFO) */
   175            Mode_BINARY = 0x1,             /*! Binary (FIFO) */
   176            Mode_COUNTING_PRIORITY = 0x2,  /*! Counting (priority-based) */
   177            Mode_BINARY_PRIORITY = 0x3     /*! Binary (priority-based) */
   178        };

   345        @DirectCall
   346        Int getCount();
   347    
   348        /*!
   349         *  ======== pend ========
   350         *  Wait for a semaphore
   351         *
   352         *  If the semaphore count is greater than zero (available), this function
   353         *  decrements the count and returns TRUE. If the semaphore count is zero
   354         *  (unavailable), this function suspends execution of the current task
   355         *  until post() is called or the timeout expires.
   356         *
   357         *  A timeout value of
   358         *  {@link ti.sysbios.BIOS#WAIT_FOREVER BIOS_WAIT_FOREVER} causes
   359         *  the task to wait indefinitely for its semaphore to be posted.
   360         *
   361         *  A timeout value of {@link ti.sysbios.BIOS#NO_WAIT BIOS_NO_WAIT}
   362         *  causes Semaphore_pend to return immediately.
   363         *
   364         *  @param(timeout)     return after this many system time units
   365         *
   366         *  @b(returns)         TRUE if successful, FALSE if timeout
   367         */
   368        @DirectCall
   369        Bool pend(UInt timeout);
   370    
   371        /*!
   372         *  ======== post ========
   373         *  Signal a semaphore.
   374         *
   375         *  Readies the first task waiting for the semaphore. If no task is
   376         *  waiting, this function simply increments the semaphore count and
   377         *  returns.  In the case of a binary semaphore, the count has a
   378         *  maximum value of one.
   379         */
   380        @DirectCall
   381        Void post();

   384         *  ======== registerEvent ========
   385         *  Register an Event Object with a semaphore
   386         *
   387         *  Ordinarily, an Event object and eventId are configured at
   388         *  Semaphore create time.
   389         *
   390         *  This API is provided so that Semaphore-using middleware
   391         *  can support implicit Event posting without having to be
   392         *  retrofitted.
   393         *
   394         *  After the Event object and eventId are registered with the
   395         *  Semaphore:
   396    
   397         *  Event_post(event, eventId) will be invoked when
   398         *  Semaphore_post(sem) is called.
   399         *
   400         *  Event_pend(event, eventId, 0, timeout) will be invoked when
   401         *  Semaphore_pend(sem, timeout) is called.
   402         *
   403         *  @param(event)                     Ptr to Event Object
   404         *  @param(eventId)                   Event ID
   405         *
   406         */
   407        @DirectCall
   408        Void registerEvent(Event.Handle event, UInt eventId);

   415         *  No task switch occurs when calling SEM_reset.
   416         *
   417         *  @a(constraints)

   419         *

   422         *
   423         *  Semaphore_reset cannot be called by a Hwi or a Swi.
   424         *
   425         *  @param(count)                   semaphore count
   426         *
   427         */
   428        @DirectCall
   429        Void reset(Int count);
